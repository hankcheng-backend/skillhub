3. Add a new `update_source_token` Tauri command in `sources.rs`:

```rust
#[tauri::command]
pub async fn update_source_token(
    db: State<'_, DbState>,
    source_id: String,
    new_token: String,
) -> Result<(), AppError> {
    let trimmed = new_token.trim();
    if trimmed.is_empty() {
        return Err(AppError::Conflict("Token cannot be empty".into()));
    }

    let keychain_key = {
        let conn = db.lock().unwrap();
        let source = Source::find_by_id(&conn, &source_id)
            .map_err(|_| AppError::NotFound(format!("Source not found: {}", source_id)))?;
        match source.keychain_key {
            Some(key) => key,
            None => {
                // Source has no keychain_key yet — create one
                let key = format!("skillhub-{}", source_id);
                Source::update_keychain_key(&conn, &source_id, &key)?;
                key
            }
        }
    };

    oauth::store_token("skillhub", &keychain_key, trimmed)?;
    Ok(())
}
```

Per pitfall 5 from research: this updates the token in-place via the existing keychain entry. No source deletion or re-creation. The source UUID is preserved.
